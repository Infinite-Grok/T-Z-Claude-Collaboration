function drawChromaticAberration(intensity) {

  // Red channel: shift left
  ctx.globalCompositeOperation = 'screen';
  ctx.fillStyle = `rgba(255, 0, 0, ${intensity * 0.5})`;

}

class Particle {
  constructor() {
    this.x = Math.random() * canvas.width;
    this.y = -50;
    this.vx = (Math.random() - 0.5) * 4;
    this.vy = Math.random() * 3 + 2;
    this.text = ['<<<<', '====', '>>>>', 'âš ', 'âœ–', 'â›”'][Math.floor(Math.random() * 6)];
    this.opacity = 1;
    this.rotation = Math.random() * Math.PI * 2;
    this.rotationSpeed = (Math.random() - 0.5) * 0.1;
  }

  update() {
    this.x += this.vx;
    this.y += this.vy;
    this.vy += 0.1; // Gravity
    this.rotation += this.rotationSpeed;
    this.opacity -= 0.01;

    // Bounce off edges
    if (this.x < 0 || this.x > canvas.width) this.vx *= -1;
  }

  draw() {
    ctx.save();
    ctx.translate(this.x, this.y);
    ctx.rotate(this.rotation);
    ctx.fillStyle = `rgba(255, 107, 107, ${this.opacity})`;
    ctx.font = '24px Courier New';
    ctx.fillText(this.text, 0, 0);
    ctx.restore();
  }

  isDead() {
    return this.opacity <= 0 || this.y > canvas.height + 50;
  }
}
